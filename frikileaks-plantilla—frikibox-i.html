id info

Uso

{{Frikibox-i
|align= alineación
|id= id texto o imagen
|id-c = id color de fondo
|id-fc = id color de fuente
|id-s = id tamaño del texto
|info = info texto
|info-c = info color de fondo
|info-fc = info color de fuente
|info-s = info tamaño del texto
|border-c = color de la caja
|border-s = grosor de la caja
}}

Ejemplo


{{Frikibox-i
|align=right
|id= Caja ID
|id-c = red
|id-fc = white
|id-s = 10
|info = Contenido caja
|info-c = yellow
|info-fc = blue
|info-s = 11
|border-c = green
|border-s = 3
}}
